LLM 大模型的数据格式：
```TypeScript
interface IWeatherDay {
  tempMax: number; // 最高温度（摄氏度）
  tempMin: number; // 最低温度（摄氏度）
  weather: string; // 天气描述
}
interface IWeather {
  area: string; // 区域
  daytime: IWeatherDay; // 白天天气
  nighttime: IWeatherDay; // 夜间天气
  clothing: string; // 穿衣建议
}

interface IAccommodation {
  name: string; // 名称
  address: string; // 地址
  feature: string; // 特色
  booking?: string; // 预定建议

  // 高德 API 获取
  price?: number; // 价格（消费）
}

interface IActivity {
  name: string; // 名称
  description: string; // 描述
  suggestedHours: string; // 建议游玩时间

  // 高德 API 信息
  ticketPriceCny: number; // 门票价格（消费）
  openingHours: string; // 开发时间
  images: [{ description: string, imgSrc: string }] // 参考图片信息
}

interface ITravel {
  day: number;
  title: string;
  waypoints: string; // 路线途径点（如:['喀纳斯', '禾木', '乌禾木']
  description: string;
  accommodation: IAccommodation[]; // 1-3个推荐住宿
  foodRecommendation: string[]; // 1-3个推荐美食
  commentTips?: string; // 特别提示、注意事项等特殊描述（该字段可选）
  activities: IActivity[]

  // 高德 API 信息
  distance: number; // 距离
  drivingHours: number; // 驾车/路程时间
}

interface ITravelPlan {
  planName: string;
  totalDays: number;
  totalDistance: number; // 大概距离
  vehicleType: string; // 出行方式
  vehicleAdvice: string; // 出行建议
  bestSeason: string; // 出行月份
  essentialItems: string[]; // 必备物品

  weather: IWeather[]; // 天气信息
  days: ITravel[]
}
```
